/** 
 * Set the table columns and content. 
 * @implNote This resets all custom value converters.
 * @param dataVector
 * @param columnIdentifiers
 */
private void setDataVector(Object[][] dataVector, Object[] columnIdentifiers) {
	this.columnIdentifiers = columnIdentifiers;
	
	TableColumnModel columnModel = table.getColumnModel();
	int numColumns = columnModel.getColumnCount();

	// Reset all columns to use the default renderer
	for (int i = 0; i < numColumns; i++) {
	    TableColumn column = columnModel.getColumn(i);
	    column.setCellRenderer(null);
	}
	
	setDataVector(dataVector);
}
